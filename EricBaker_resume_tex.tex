\newcommand{\resumeItem}[3]{
  \item[]\small{
    \textbf{#1}{: #2 \vspace{-2pt}}
  }
}

\newcommand{\resumeItemWithTag}[3]{
  \item\small{
    \underline{#1}{: #2    }
    \cvtaglist{#3}
  }
}

\newcommand{\resumeBullet}[1]{
  \item[]\small{
    \vspace{-4pt}
    {#1}
    \vspace{-2pt}
  }
}

\newcommand{\resumeBulletItalic}[1]{
  \item[]\small{
    \vspace{-2pt}
    \textit{#1}
    \vspace{-1pt}
  }
}

      #3 & \textit{\small #4} \\
