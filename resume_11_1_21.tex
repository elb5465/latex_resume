% Custom commands
\newcommand{\resumeItem}[3]{
  \item[]\small{
    \textbf{#1}{: #2 \vspace{-2pt}}
  }
}
\newcommand{\resumeItemWithTag}[3]{
  \item\small{
    \textbf{#1}{: #2    }
    \cvtaglist{#3}
  }
}
\newcommand{\resumeBullet}[1]{
  \item[]\small{
    \vspace{-4pt}
    {#1}
    \vspace{-2pt}
  }
}

\newcommand{\resumeBulletItalic}[1]{
  \item[]\small{
    \vspace{-2pt}
    \textit{#1}
    \vspace{-1pt}
  }
}

\newcommand{\languageUsedWithBar}[1]{ \textbar \enspace  \textcolor{Gray}{\footnotesize{#1}} }
\newcommand{\languageUsed}[1]{ \enspace  \textcolor{Gray}{\scriptsize{#1}} }

\newcommand{\cvtag}[1]{%
  \tikz[baseline]\node[anchor=base,draw=darkGrey!70,rounded corners=0.5ex,inner xsep=1ex,inner ysep =1.5ex,text height=1ex,text depth=0ex]{#1};
}
\newcommand{\cvtaglist}[1]{
    \footnotesize{\foreach \n in {#1}{\cvtag{\n}}} 
}

\newcommand{\resumeSubheading}[5]{
  \vspace{-1pt}\item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \textbf{#1} & \textit{\small#2} \\ 
      #3 & \textit{\small #4} \\ 
    \end{tabular*} \\
    % \vspace{0.1cm}    %give space between summary and job title
    \hspace{0.5cm}#5  %indent summary 
    \vspace{-0.225cm}   %bring following text closer
}

\newcommand{\studyAbroadEntry}[4]{
  \vspace{-1pt}\item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \hspace{0.45cm}\circ\textbf{ \small#1} \small\textit{#3} & \textbf{\small#2} \textit{\small #4} \\
    \end{tabular*}\vspace{-5pt}
}
